def plot_filter(filter_taps, fs):
    w,h = sig.freqz(filter_taps, fs=fs)
    plt.figure()
    plt.plot(w,20*np.log10(np.abs(h)**2))
    plt.xlabel('$f [Hz]$')
    plt.ylabel(r'$\left|\tilde{H}\left(f\right)\right|^{2}$')
    plt.show()

def papr(x):
    x = np.abs(x)
    return x.max()/np.sqrt(np.mean(np.abs(x)**2))
